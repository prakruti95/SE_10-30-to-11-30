#include<stdio.h>
#include<conio.h>

void main()
{
    int i, j, spaces;

    for(i = 1; i <= 5; i++)
    {
     
        for(j = 1; j <= i; j++)
        {
            printf("%d", j);
        }

        
        spaces = 2 * (5 - i);
        for(j = 1; j <= spaces; j++)
        {
            printf(" ");
        }

  
        for(j = i; j >= 1; j--)
        {
            printf("%d", j);
        }

        printf("\n");
    }

    getch();
}
